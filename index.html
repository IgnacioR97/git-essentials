Copy this code to use github with vs code

/* Step 1: remove existing git reposit */

rm -rf .git

/* Step 2: create an empty git repository */

git init

/* Step 3: adds new files the git area */

git add

/* Step 4: adds the entire project to git area */

git add .

/* Step 5: creates a commit with a message */

git commit -m "message"

/* Step 6: push all changes */

git remote add origin git@github.com:your-profile/repo-name.git

git branch -M main

git push -u origin main

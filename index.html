Copy this code to use github with vs code

/* Step 1: remove existing git reposit */

rm -rf .git

/* Step 2: create an empty git repository */

git init

/* Step 3: clone your github repository */

git clone (url-to-repository)

/* Step 4: add entire project */

git add .

/* Step 5: commit changes with message */

git commit -m "message"

/* Step 6: push all changes */

git push
